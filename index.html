    Tip: save blob to image

        const blobToImage = blob => {
      const $img = document.createElement('img')
      
      $img.src = URL.createObjectURL(blob)
      $img.onload = () => URL.revokeObjectURL($img.src)
      
      document.querySelector('.output').append($img)
    }
